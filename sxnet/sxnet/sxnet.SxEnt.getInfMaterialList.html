SxEnt.getInfMaterialList メソッド 

指定要素群の材質情報を問い合わせます。
このメソッドはiCAD SX ビューア／iCAD SX 組立支援の場合でも実行可能です。

[Visual Basic]
Public Shared Function getInfMaterialList( _
   ByVal entlist As SxEnt() _
) As SxInfMaterial()[)
[C#]
public static SxInfMaterial[][] getInfMaterialList(
   SxEnt[] entlist

パラメータ

entlist
問合せを行う要素(パーツ／セグメント)の配列

戻り値

材質情報の多段階配列(ジャグ配列)

解説

指定要素群に付加された材質情報を問い合わせます。
entlist[]には、同一WF内のパーツ／セグメントを指定して下さい。
パーツを指定した場合、パーツ内の要素に付加されている材質情報を全て取得 しますが、子パーツ内の要素に付加されている情報は取得しません。
指定セグメント、パーツ内の要素に材質が付加されていない場合、 該当する配列の要素にはnull(VBの場合、Nothing)を返答します。

[C#] 

// パーツの取得
SxWF wf = SxWF.getActive();
SxEntPart[] partlist = wf.getPartList();

// パーツの材質情報を一括取得。
SxInfMaterial[][] matlist = SxEnt.getMaterialList(partlist);

// パーツ毎に、材質名称を表示する。
for ( int i = 0; i < partlist.Length; i++ )
{
    // パーツ名の表示
    SxInfPart inf = partlist[i].getInf();
    Console.WriteLine("No " + i+1 + ":" + inf.name);

    // 材質情報の表示
    if ( matlist[i] == null )
    {
        // パーツには材質が付加されてない。
        Console.WriteLine("  None.");
    }
    else
    {
        // パーツ内要素に付加された、全ての材質名称を表示
        for ( j = 0; j < matlist[i].Length; j++ )
        {
            Console.WriteLine("  " + matlist[i][j].name);
        }
    }
}

    Dim wf As SxWF = SxWF.getActive()

 ' パーツの材質情報を一括取得
 Dim matlist As SxInfMaterial()() = SxEnt.getInfMaterialList(plist)

 ' パーツ毎に、材質名称を表示する。
    For i As Integer = 0 To plist.Length - 1
     ' パーツ名の表示
        Dim inf As SxInfPart = plist(i).getInfDetail()
        Console.WriteLine("No " & (i + 1) & ":" & inf.name)
 
     ' 材質情報の表示
        If matlist(i) Is Nothing Then
         ' パーツには材質が付加されていない。
            Console.WriteLine("  None.")
        Else
         ' パーツ内要素に付加された、全ての材質名称を表示
            For j As Integer = 0 To matlist(i).Length - 1
                Console.WriteLine("  " & matlist(i)(j).name)
            Next j
        End If
    Next i
